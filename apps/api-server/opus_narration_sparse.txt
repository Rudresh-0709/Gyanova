## Agent Prompt: Fix Sparse Template Pipeline

**Objective:** Fix the slide generation pipeline so that sparse templates ("Title Card", "Image and text", "Text and image") generate correct content, pass validation, and trigger their existing sparse narration code properly.

**Problem Summary:**
The pipeline has a two-part failure for sparse templates:
1. The GyML generator doesn't know what blocks to produce for sparse templates, so the LLM falls back to generic paragraphs
2. The validator requires a primary block, which sparse templates intentionally don't have
3. This causes validation failure → retry → fallback to hardcoded bullets → the existing sparse narration code in `generate_narration()` never activates on correct content

**The existing sparse narration code in `content_generation_node.py` (around line 500) already handles sparse templates correctly — it just never receives properly generated content.**

```python
SPARSE_TEMPLATE_SCHEMAS = {
    "Title card": {
        "required_blocks": ["intro_paragraph"],
        "optional_blocks": ["annotation_paragraph"],
        "forbidden_blocks": ["smart_layout", "bullet_list", "table", "numbered_list"],
        "instruction": (
            "Generate ONLY a single intro_paragraph block. "
            "This is a title card — one powerful engagement hook sentence or short paragraph. "
            "No lists, no tables, no smart_layout, no structured content. Just the intro."
        ),
        "max_blocks": 2,
    },
    "Image and text": {
        "required_blocks": ["image", "rich_text"],
        "optional_blocks": ["annotation_paragraph"],
        "forbidden_blocks": ["smart_layout", "bullet_list", "table"],
        "instruction": (
            "Generate an image block with descriptive alt text and a clear visual description, "
            "followed by a rich_text block that explains or narrates the visual content. "
            "No smart_layout, no lists, no tables."
        ),
        "max_blocks": 4,
    },
    "Text and image": {
        "required_blocks": ["intro_paragraph", "image"],
        "optional_blocks": ["rich_text", "annotation_paragraph"],
        "forbidden_blocks": ["smart_layout", "bullet_list", "table"],
        "instruction": (
            "Generate an intro_paragraph that explains the concept first, "
            "then an image block as visual reinforcement with descriptive alt text. "
            "No smart_layout, no lists, no tables."
        ),
        "max_blocks": 4,
    },
}

def get_sparse_template_schema(template_name):
    """
    Returns the sparse template schema if template_name is a sparse template.
    Returns None if it's a standard template.
    Case-insensitive matching.
    """
    if not template_name:
        return None
    normalized = template_name.strip().lower()
    for key, schema in SPARSE_TEMPLATE_SCHEMAS.items():
        if key.lower() == normalized:
            return schema
    return None

def is_sparse_template(template_name):
    """Quick check if a template is sparse."""
    return get_sparse_template_schema(template_name) is not None
```

**Important:** This dictionary is the single source of truth. Both the generator and validator import from here. Do NOT duplicate these definitions.

```python
from app.services.node.narration_techniques import get_sparse_template_schema, is_sparse_template

def _validate_sparse_template(raw_content, template_name):
    """
    Validates content generated for sparse templates.
    
    Sparse templates don't need a primary block. Instead, we verify:
    1. All required blocks for this template type are present
    2. No forbidden blocks are present
    3. Total block count doesn't exceed the maximum
    
    Returns:
        validated content dict if valid, None if validation fails
    """
    schema = get_sparse_template_schema(template_name)
    if not schema:
        print(f"    ⚠ No sparse schema found for '{template_name}'")
        return None

    # Extract block types from the generated content
    # Adapt this to however your content structure works
    blocks = []
    if isinstance(raw_content, dict):
        # If content is a dict with a 'blocks' key
        content_blocks = raw_content.get('blocks', [])
        if isinstance(content_blocks, list):
            blocks = [b.get('type', '') for b in content_blocks if isinstance(b, dict)]
        # Also check top-level keys that might be block types
        for key in raw_content.keys():
            if key not in ('blocks', 'metadata', 'template_name', 'primary_block_index'):
                blocks.append(key)
    
    if not blocks:
        print(f"    ✗ Sparse validation: No blocks found in content")
        return None

    print(f"    ℹ Sparse validation for '{template_name}': found blocks = {blocks}")

    # Check required blocks exist
    missing = []
    for required in schema['required_blocks']:
        if required not in blocks:
            missing.append(required)
    
    if missing:
        print(f"    ✗ Sparse template '{template_name}' missing required blocks: {missing}")
        return None

    # Check forbidden blocks absent
    violations = []
    for forbidden in schema['forbidden_blocks']:
        if forbidden in blocks:
            violations.append(forbidden)
    
    if violations:
        print(f"    ✗ Sparse template '{template_name}' contains forbidden blocks: {violations}")
        return None

    # Check max blocks
    if len(blocks) > schema['max_blocks']:
        print(f"    ⚠ Sparse template '{template_name}' has {len(blocks)} blocks (max: {schema['max_blocks']}), trimming...")
        # Optionally trim excess blocks here instead of failing
        # For now, still accept it but warn

    print(f"    ✓ Sparse template '{template_name}' validated successfully")
    return raw_content
```

```python
def _construct_sparse_fallback(template_name, topic):
    """
    Constructs minimal valid content for a sparse template
    when generation and retry both fail.
    """
    schema = get_sparse_template_schema(template_name)
    if not schema:
        return None

    normalized = template_name.strip().lower()
    
    if normalized == "title card":
        return {
            "blocks": [
                {
                    "type": "intro_paragraph",
                    "content": f"Exploring {topic}"
                }
            ],
            "primary_block_index": None,
            "template_name": template_name,
        }
    elif normalized == "image and text":
        return {
            "blocks": [
                {
                    "type": "image",
                    "alt": f"Visual representation of {topic}",
                    "description": f"An illustration showing key aspects of {topic}"
                },
                {
                    "type": "rich_text",
                    "content": f"Let's explore {topic} through this visual guide."
                }
            ],
            "primary_block_index": None,
            "template_name": template_name,
        }
    elif normalized == "text and image":
        return {
            "blocks": [
                {
                    "type": "intro_paragraph",
                    "content": f"Understanding {topic} starts with the fundamentals."
                },
                {
                    "type": "image",
                    "alt": f"Visual representation of {topic}",
                    "description": f"An illustration reinforcing the concept of {topic}"
                }
            ],
            "primary_block_index": None,
            "template_name": template_name,
        }
    
    return None
```
